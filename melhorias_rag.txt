## Exemplos de Implementação

### 1. Chunking Adaptativo
```python
def chunking_adaptativo(texto):
    # Detecta seções/títulos
    secoes = re.split(r'\n#{1,6}\s+', texto)
    
    chunks = []
    for secao in secoes:
        # Tamanho adaptativo baseado no conteúdo
        tamanho_chunk = min(len(secao), 1000)
        
        # Preserva contexto semântico
        if len(secao) > tamanho_chunk:
            # Divide em sentenças
            sentences = nltk.sent_tokenize(secao)
            current_chunk = ""
            
            for sent in sentences:
                if len(current_chunk) + len(sent) <= tamanho_chunk:
                    current_chunk += sent + " "
                else:
                    chunks.append(current_chunk.strip())
                    current_chunk = sent + " "
            
            if current_chunk:
                chunks.append(current_chunk.strip())
        else:
            chunks.append(secao.strip())
    
    return chunks
```

### 2. Hybrid Search
```python
def hybrid_search(query, top_k=5):
    # Busca por embeddings
    query_embedding = hf.embed_query(query)
    vector_results = db.search_similar_documents(query_embedding, top_k=top_k*2)
    
    # Busca por BM25
    bm25_results = db.search_bm25(query, top_k=top_k*2)
    
    # Combina resultados
    combined_results = []
    seen_ids = set()
    
    for result in vector_results + bm25_results:
        if result['id'] not in seen_ids:
            seen_ids.add(result['id'])
            combined_results.append(result)
    
    # Reranking
    reranked_results = rerank_results(query, combined_results[:top_k])
    
    return reranked_results
```
